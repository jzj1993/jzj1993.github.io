  2. define( 'TPL_DIR', get_template_directory() );

  3. // Log信息输出

  4. function wplog( $str = '', $tag = '' ) {

  5. $split = ( $tag=='' ) ? '' : ":\t";

  6. file_put_contents( TPL_DIR.'/wp.log', $tag . $split . $str . "\n", FILE_APPEND );

  7. }
